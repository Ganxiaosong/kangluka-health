const CLOUD_BASE = 'https://kangluka-d0gjhx9wya86f10f9.service.tcloudbase.com'; async function handleLogin() { const email = document.getElementById('emailInput').value.trim(); const password = document.getElementById('passwordInput').value; if (!email || !password) { showToast('请输入邮箱和密码'); return; } try { const res = await fetch(CLOUD_BASE + '/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email, password }) }); const result = await res.json(); if (!result.success) { showToast(result.message); return; } currentUser = result.data; DB.set('kangluka_user', currentUser); showToast(result.message); showHome(); // 如果是新用户，提示完善资料 if (!currentUser.name) { openEditProfile(); } } catch (err) { showToast('登录失败，请检查网络'); console.log(err); } } async function loadUserData() { if (!currentUser) return; try { const res = await fetch(CLOUD_BASE + '/getRecords?uid=' + currentUser.uid); const result = await res.json(); if (result.success) { currentUser = { ...currentUser, ...result.data.user, records: result.data.records }; renderData(); } } catch (err) { console.log('加载数据失败:', err); } }
